import cairo


def generate_surface(elements=1):
    surface = cairo.RecordingSurface(cairo.CONTENT_COLOR_ALPHA, None)
    context = cairo.Context(surface)
    context.set_source_rgb(0, 0, 0)
    context.set_line_width(.5)

    for i in range(elements):
        context.move_to(i+1, 0)
        context.line_to(i, 10)
        context.stroke()

    return surface


def paint_surface(context, to_draw, x=0, y=0):
    context.save()
    context.set_source_surface(to_draw, x, y)
    context.paint()
    context.restore()


def draw_test(surface_count=15, offset=20):
